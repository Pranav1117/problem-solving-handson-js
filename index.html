  1. Find the Smaller Angle

    2. Answer:

    var Minimal_Angle = (h, m) => 
{
    const minuteAngle = m/ 60 * 360;
    const hourAngle = (h% 12 + m / 60 ) * 360 / 12;
    let ans = Math.abs(minuteAngle - hourAngle);
    return ans;
       
};

    var Perfect_Check = (N) => 
{
  var c=0;
var temp = 0;
for(var i=1;i<=N/2;i++)
{ 
  if(N%i === 0)
  { 
    temp += i;
  }
}
  if(temp === N && temp !== 0)
  { 
    c ='YES';
  } 
    else 
    { 
      c ='NO';
    } 
  
  return c;
};
